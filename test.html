html2canvas(document.querySelector("#capture")).then(canvas => { // 이미지 데이터 URL 생성 const imageURL = canvas.toDataURL("image/png"); // a 엘리먼트 생성 const link = document.createElement('a'); link.href = imageURL; link.download = 'captured-image.png'; // 저장될 파일명 지정 // a 태그 클릭 트리거 document.body.appendChild(link); link.click(); // 링크 제거 document.body.removeChild(link); });
